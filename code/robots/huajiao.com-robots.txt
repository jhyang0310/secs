function _hj_get_cookie(cname) {
  var temp, cookies = document.cookie.split(";");
  for (var i = 0; i < cookies.length; i++) {
    temp = cookies[i].split("=");
    if (temp[0].replace(/^\s\s*/, '').replace(/\s\s*$/, '') == cname) {
    	var v = unescape(temp[1]);
    	v = eval('('+v+')');
    	return v;
    }
  }
}

}
