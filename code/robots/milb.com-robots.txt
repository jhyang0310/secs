	(function($){

		initSlider = function(){
			var lastSlideGroupIndex,
			totalPages,
			totalSlides,
			isLastPage,
			isFirstPage;

			$('.slider1').bxSlider({
					autoControls: true,
					displaySlideQty: 6,
					moveSlideQty: 6,
					speed: 600,
					infiniteLoop : false,
					onBeforeSlide : function(currentSlideNumber, totalSlideQty, currentSlideHtmlObject){
						//total page count
						totalPages = Math.ceil(totalSlideQty/this.displaySlideQty);

						totalSlides = totalSlideQty;

						//index of first item in last slide group
						//if moveSlideQty = 8 and totalPages = 3
						//indices are then [0,8,16] where slide index begins at zero incrementing by 8 or moveSlideQty
						//so lastSlideGroupIndex = 16
						lastSlideGroupIndex = (totalPages * this.displaySlideQty) - this.displaySlideQty;
		
						//figure out if the next page is the last page
						// 8 + 8 === 16
						isLastPage = (currentSlideNumber + this.displaySlideQty === lastSlideGroupIndex) ? true : false;

						//if last page offset the move amount otherwise use default
						//this will help stop the carousel on the final page
						if(isLastPage){
								this.moveSlideQty = totalSlideQty - ((totalPages - 1) * this.displaySlideQty);
						}else{
								this.moveSlideQty = this.displaySlideQty;
						}
					},
					onAfterSlide : function(currentSlideNumber, totalSlideQty, currentSlideHtmlObject){
						isFirstPage = currentSlideNumber === 0 ? true : false;
						totalSlides = totalSlideQty;
						if(isFirstPage){
								this.onPrevSlide();  
						}
					},
					onNextSlide : function(currentSlideNumber, totalSlideQty, currentSlideHtmlObject){
						if(isLastPage){
								$(".video-carousel .bx-next").addClass("hideButton");
						}else{
								$(".video-carousel .bx-next").removeClass("hideButton");
						}
								$(".video-carousel .bx-prev").removeClass("hideButton");
					},
					onPrevSlide:function(currentSlideNumber, totalSlideQty, currentSlideHtmlObject){
						if(isFirstPage){
								$(".video-carousel .bx-prev").addClass("hideButton");
						}else{
								$(".video-carousel .bx-prev").removeClass("hideButton");
						}
						$(".video-carousel .bx-next").removeClass("hideButton");

						//hide next button if total slide count is less thean display count
						if(totalSlides < this.displaySlideQty){
								$(".video-carousel .bx-next").addClass("hideButton");
						}
					}
			}).goToFirstSlide();
		};

		$(function(){
			initSlider();
		});	
	}(jQuery))
