        $divBelowNavbar.css('padding-top', 65);
    } else {
        $navbarTransparent.removeClass('navbar-fixed-top');
        $navbarTransparent.removeClass('bg-danger');
        $navbarTransparent.addClass('navbar-light');
        $navbarTransparent.removeClass('navbar-dark');
        $divBelowNavbar.css('padding-top', 10);
    }

    if ($(document).width() > 768) {
        if (path === '') {
            if ($(window).scrollTop() > 800) {
                $('.nav-item-download-app').show();
                $('#nav-link-download-small').hide();
            } else {
                $('.nav-item-download-app').hide();
                $('#nav-link-download-small').hide();
            }
        } else {
            if ($(window).scrollTop() > 10) {
                $('.nav-item-download-app').show();
                $('#nav-link-download-small').hide();
            } else {
                $('.nav-item-download-app').hide();
                $('#nav-link-download-small').hide();
            }
        }
    }

    if ($(document).width() < 768) {
        if (path === '') {
            if ($(window).scrollTop() > 300) {
                $('#nav-link-download-small').show();
            } else {
                $('#nav-link-download-small').hide();
            }
        } else {
            if ($(window).scrollTop() > 10) {
                $('#nav-link-download-small').show();
            } else {
                $('#nav-link-download-small').hide();
            }
        }

    }
});
