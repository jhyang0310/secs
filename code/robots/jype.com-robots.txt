	function tLinkFormat(page) {

		var flDiv = location.href;
		var MDiv = flDiv.split("?LDiv=");
		var QDiv = flDiv.split("&LDiv=");	
		
		if (MDiv[1] == undefined)
		{
			MDiv[1] = QDiv[1];
			
			if (MDiv[1] == undefined) 
			{
				MDiv[1] = "KO";
			}
		}
		
		if (page == "Main"){
		  location.href = "/Main.aspx?LDiv="+ MDiv[1];
		}
		else if (page == "AboutJYP"){
		  location.href = "/company/AboutJYP.aspx?LDiv="+ MDiv[1];
		}
		else if (page == "History") {
		  location.href = "/company/History.aspx?LDiv="+ MDiv[1];		
		}

		}
		else if (page == "Artist") {
		  location.href = "/common/ArtistMain.aspx?LDiv="+ MDiv[1];		
		}

		  location.href = "/audition/Audition.aspx?LDiv="+ MDiv[1];
