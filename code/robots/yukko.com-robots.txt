        function closeLastVisit(buton, stoklink) {
            var codes = new Array();
            if (stoklink != '') {
                var pCode = $.cookie('_lastvisited');
                if (pCode && pCode != '') {
                    codes = pCode.split('|');
                }
                for (var i = 0; i < codes.length; i++) {
                    var items = codes[i].split(';');
                    if (items && items.length > 2 && items[1] == stoklink) {
                        codes.splice(i, 1);
                        $(buton).parent().remove();
                        $('.lastvisited-items').tinycarousel({
                            start: $.tiny.current, infinite: false,
                            animationTime: 400
                        });

                    }
                    //else if(items && items.length < 3 ) {
                    //    codes.splice(i, 1);
                    //}
                }
                $.cookie('_lastvisited', codes.join('|'), { expires: 30, path: '/' });
                var valid = false;
                for (var i = 0; i < codes.length; i++) {
                    var items = codes[i].split(';');
                    if (items && items.length > 2) {
                        valid = true;
                        break;
                    }
                }
                if (!valid) {
                    $('#lastvisited-container').hide();
                }
            }
        }
