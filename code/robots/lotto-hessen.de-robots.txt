  <script type="text/javascript">
  
      if(typeof etCommerce != 'undefined') {
      etCommerce.debugMode = true;
    }
      
  if(typeof et_cc_mode != 'undefined') {
    if(typeof etr_adding != 'undefined' && etr_adding == ''){
      etr_adding = '--'
    }
    sendCampaignControlEvent(et_cc_mode)
  }

  function sendCampaignControlEvent(aETCCMode) {
    
    if(typeof etr_variants == 'undefined') {
      etr_variants = '';
    }
    
    if(typeof etr_duration == 'undefined') {
      etr_duration = '--';
    }
    
    if(typeof etCommerce == 'undefined') {
      return;
    }
  
    if(aETCCMode == "prereceipt") {
      
      if(typeof et_basket_id == 'undefined') {
        return;
      }
  
      /* Campaign Control Tracking: Insert To Basket auf Vorquittung */
      
      var product =
      {
        id      :etr_id,
        name    :etr_name,
        category:[etr_game,etr_gamename,etr_day],
        price   :etr_price,
        currency:'EUR',
        variants:
        {
          'Zusatzspiel' : checkVariants(etr_adding),
          'Laufzeit' : checkVariants(etr_duration),
          'Spezial' : checkVariants(etr_variants)
        }
      };
      etCommerce.sendEvent('insertToBasket', product, 1, et_basket_id);
      
      $('.et_cc_removeFromBasket').click(function(){
        /* Campaign Control Tracking: Remove From Basket auf Vorquittung */
        etCommerce.sendEvent('removeFromBasket', product, 1, et_basket_id);
      });
    
    } else if (aETCCMode == "receipt") {
      
      if(typeof et_basket_id == 'undefined') {
        return;
      }

      if(et_basket_id == '') {
        et_basket_id = 'Warenkorb 1';
      }
    
      var tProducts = [];
      for(i = 0; i < etr_id.length; i++)
      {
        var tProduct =
        {
          product:
          {
            id      :etr_id[i],
            name    :etr_name[i],
            category:[etr_game[i],etr_gamename[i],etr_day[i]],
            price   :etr_price[i],
            currency:'EUR',
            variants:
            {
              'Zusatzspiel' : checkVariants(etr_adding[i]),
              'Laufzeit' : checkVariants(etr_duration[i]),
              'Spezial' : checkVariants(etr_variants[i])
            }
          },
          quantity: etr_quantity[i]
        };
        tProducts.push(tProduct);
      }
      
      /* Campaign Control Tracking: Order auf Quittung */
      
      var basket =
      {
        id: et_basket_id,
        products: tProducts
      }
      
      var order =
      {
        orderNumber:getGlobalIdentifier(),
        status:'sale',
        orderPrice:etr_totalprice,
        currency:'EUR',
        basket:basket,
        customerGroup:'Stammkunde',
        paymentMethod:etr_payment
      }
      etCommerce.sendEvent('order',order,'Warenkorb');
    
    } else if(aETCCMode == "register") {
      var guid = getGlobalIdentifier();
      /* Campaign Control Tracking: Order auf Registrierung */
      console.log("register tracking");
      
      var basket =
      {
        id:guid,
        products:[
        {
          product:
          {
            id      :'REG',
            name    :'Registrierung',
            category:['Registrierung'],
            price   :'1.00',
            currency:'EUR',
            variants:{'Reg':'Reg'}
          },
          quantity:1
        }]
      };
  
      var order =
      {
        orderNumber:Date.now().toString(),
        status:'lead',
        orderPrice:'1.00',
        currency:'EUR',
        basket:basket,
        customerGroup:'18'
      };
      etCommerce.sendEvent('order',order,'Registrierung');
      
    }
  }
  
  /* Helper functions */
  
  function getGlobalIdentifier() {
    function s4() {
      return Math.floor((1 + Math.random()) * 0x10000).toString(16).substring(1);
    }
    return s4() + s4() + '-' + s4() + '-' + s4() + '-' + s4() + '-' + s4() + s4() + s4();
  }
  
  function checkVariants(aValue) {
    if (aValue == '')
      aValue = '-';

    return aValue;
  }
