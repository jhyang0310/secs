var entityMap = {
  '&': '&amp;',
  '<': '&lt;',
  '>': '&gt;',
  '`': '&#x60;',
  '=': '&#x3D;'
};

function escapeHtml (string) {
  return String(string).replace(/[&<>`=]/g, function (s) {
    return entityMap[s];
  });
}

function showWaitMe() {

    var waitMessage = 'Thank you for reporting this broken link.  Please wait while your report is submitted...'

    //Thanks to https://github.com/vadimsva/waitMe/issues/2 for how to use Modernizr as a fall-back for browsers that don't support CSS3 animations
    if(!Modernizr.cssanimations || !Modernizr.csstransitions) {
      $("#submit").attr("disabled", "disabled");
      $("#waitme-fallback-message").html("<strong>" + waitMessage + "</strong>");
    } else {
      $('#bd').waitMe({
        effect: 'progressBar',
        text: waitMessage,
        maxSize: '',
        color: '#000',
        bg: 'rgba(255,255,255,0.9)',
        maxSize: '',
        textPos: 'vertical',
        source: '',
      });
    }
}
