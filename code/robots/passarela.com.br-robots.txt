    function verificaEntregaGarantida() {
        resultadoPositivo = 0;
        var cep = document.getElementById("input-entrega-garantida").value;
        cep = cep.replace(/[^0-9]/g,'');

        for (i = 0; i < tabelaEntregaGarantida.length; i++) {
            for (c = 0; c < tabelaEntregaGarantida[i].length; c++) {
                if ( cep >= parseInt( tabelaEntregaGarantida[i][c].faixaInicio.replace(/[^0-9]/g,'') ) && cep <= parseInt( tabelaEntregaGarantida[i][c].faixaFim.replace(/[^0-9]/g,'') ) ) {
                    exibePositivo(cep, tabelaEntregaGarantida[i][c].cidade, tabelaEntregaGarantida[i][c].estado, i);
                    resultadoPositivo = 1;
                    break;
                } 
            } 
        }
        if ( resultadoPositivo == 0 ) {
            exibeNegativo();
        }
    }

  function exibePositivo(cep, cidade, estado, tabelaEntregaGarantida) {
        document.getElementById("retorno").style.display = "block";
        document.getElementById("retorno").innerHTML= "<div class='alert alert-success'><strong>Que bom!</strong> Seu CEP tem entrega garantida.*</div>";
        document.getElementById("conteudo").style.display = "block";
        document.getElementsByClassName("cep")[0].innerHTML = cep.substring(0, 2) + "." + cep.substring(2, 5) + "-" + cep.substring(5, 8);
        document.getElementsByClassName("cidade")[0].innerHTML = cidade;
        document.getElementsByClassName("estado")[0].innerHTML = estado;

        if ( tabelaEntregaGarantida == 0 ) {
            var liberado = "10/10 às 20h00";
        }
        if ( tabelaEntregaGarantida == 1 ) {
            var liberado = "09/10 às 20h00";
        }
           document.getElementsByClassName("liberado")[0].innerHTML = liberado;
    }
    function exibeNegativo() {
        document.getElementById("retorno").style.display = "block";
        document.getElementById("retorno").innerHTML= "<div class='alert alert-danger'><strong>Que pena!</strong> Seu CEP não tem entrega garantida.*</div>";    
        document.getElementById("conteudo").style.display = "none";
    }

						function verificaEntregaTurbo() {
						    resultadoPositivo = 0;
						    var cep = document.getElementById("input-entrega-turbo").value;
						    cep = cep.replace(/[^0-9]/g,'');
						
						    if ( cep.length == 8 ) {
						        var i = 0;
						        for (c = 0; c < tabela[i].length; c++) {
						            if ( cep >= parseInt( tabela[i][c].faixaInicio.replace(/[^0-9]/g,'') ) && cep <= parseInt( tabela[i][c].faixaFim.replace(/[^0-9]/g,'') ) ) {
						                exibePositivoTurbo();
						                resultadoPositivo = 1;
						                break;
						            } 
						        } 
						        if ( resultadoPositivo == 0 ) {
						            exibeNegativoTurbo("<strong>Que pena!</strong> Não conseguimos te atender com  a Entrega Turbo Hoje, mas temos a opção de Entrega Expressa com um preço ótimo, aproveite!");

						    } else {
						        exibeNegativoTurbo("<strong>CEP inválido!</strong> Verifique se você digitou corretamento seu CEP.");
						    }
						}
