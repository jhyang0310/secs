<script type="text/javascript">
var notificationShutdownTimer;
var notificationShutdownKey = "OPENUP_NOT_SHUT";
var notificationShutdownGate = 30 * 1000;
var notificationShutdownOption = {
	path : '/',
	expires : 1
}
var notificationShutdownStartTimeMillis = 1474243200000; 

$(function(){
	
	var executeNotificationShutDown = function() {
		//尚未到開始時間則離開
		if(new Date().getTime() < notificationShutdownStartTimeMillis) {
			return;
		}
		
		var notificationShutdown = $.cookie(notificationShutdownKey);
		if(!notificationShutdown) {
			$.cookie(notificationShutdownKey, $.now(), notificationShutdownOption);
		}
		if(notificationShutdown == '-1') {
			$.cookie(notificationShutdownKey, -2, notificationShutdownOption);
		}
		
		notificationShutdownTimer = setInterval(function(){
			
			var val = $.cookie(notificationShutdownKey);
			if(!val) val = $.now();
			else val = parseInt(val);
			
			if(val < 0) {
				clearInterval(notificationShutdownTimer);
				return;
			}
			
			if(($.now() - val) < notificationShutdownGate) {
				return;
			}
			
			$("#notificationShutdownDialog").dialog({
				modal : true,
				autoOpen : true,
				width: 500,
				maxWidth: 600,
				title : "",
				open : function() {
					$('.ui-widget-overlay').bind('click',function() {
						$(this).siblings('.ui-dialog').find('.ui-dialog-content').dialog('close');
					});
				}
			});
			
			$.cookie(notificationShutdownKey, -2, notificationShutdownOption);
			clearInterval(notificationShutdownTimer);
			return;
		}, 500);
	}
	
	$.ajax({
		url : '/api/notification/shutdown',
		success : function(resp) {
			if(resp.result == null) {
				return;
			}
			$('#notificationShutdownDialog').html(resp.result);
			executeNotificationShutDown();
		}
	});
	
});
