  function validateEmail() {
  var email = $('#emailtext').val().trim();
  if(email==''){
     alert('Please enter E-mail !');
  }else{
    
 
  var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
  if (re.test(email)) {
    addSubscription();
  } else {
    alert('Not a Valid E-mail !')
    $("#emailtext").val('');
    $("#ico-mail").removeClass("attivo");
  }
   }
}


function addSubscription() {
  var emailText = $('#emailtext').val();
  $.ajax({
    type: "POST",
    url: "ajaxPage",
    data: "fun=SubscribeUs&email_id=" + emailText,
    success: function (data, success) {
      if (data == 1) {
        alert('Thank You, You have been successfully subscribed.');
      } else if (data == 3) {
        alert('E-mail already exist!')
      }
      $("#emailtext").val('');
      $("#ico-mail").removeClass("attivo");

    },
    error: function (xml, status, e) {
    }
  });
}
