         $("#language-list li a").click(function() {

            var $this = $(this);

            var _lang = $this.data("lang");

            var _region = $this.data("region");

            var redirectionUrl = "";

            var currentPathName = window.location.pathname;

            var languageList = ["es","en", "de"];

            var languageFromUrl = "";

            for(var l = 0; l < languageList.length; l++){
                if(currentPathName.indexOf("/"+languageList[l]) != -1){
                    languageFromUrl = languageList[l];
                }
            }

            var $list = $this.parents("li");

            var updatedPath = "";

            if(_region && _region == "in"){
                if(languageFromUrl){
                    updatedPath  = currentPathName.replace("/"+ languageFromUrl, "");
                }
            }else{
                if(!languageFromUrl){
                    updatedPath = "/"+_lang + currentPathName;
                }else if(_lang != languageFromUrl){
                    updatedPath  = currentPathName.replace(languageFromUrl, _lang);
                }
            }

            if(!updatedPath){
                updatedPath = "/";
            }

            Mettl.AJAX({
                method:'POST',
                url:"/setPreferredUserLanguage",
                data:{preferredLanguage: _lang, region: _region},
                success: function (result) {
                     window.location.href = updatedPath;
                }, error: function(error){
                    console.log("error in changing language" + error.message);
                }
            });

        });

        var $bc = $(".breadcrumb");

        if($ln.length > 0){
            $ln.affix({offset: {top: $(".navbar").outerHeight(true), bottom :1000,} });
        }

    });
