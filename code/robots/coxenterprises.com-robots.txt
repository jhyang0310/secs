			/* IE6 dropdownPositioning */
			function doPositionDropdown( obj ) {

				var arVersion = navigator.appVersion.split("MSIE")
   				var version = parseFloat(arVersion[1])		
				if ((version >= 5.5) && (version < 7 ) && (document.body.filters)) {
					
					var w = obj.offsetWidth;					
					
					var uls = obj.getElementsByTagName("UL");
					var n = 0;				

					while ( uls[n].className != "Dropdown" ) n++;
					uls[n].style.marginLeft = -1 * w;

				}	

			}

			function doPositionMega ( obj ) {

				

				if ( navigator.appName.toLowerCase() == "netscape" ) {

					var mdd = obj.getElementsByTagName("DIV");

					if ( mdd.length > 0 ) {
						//alert ( mdd.length );
						var n = 0;
						while ( mdd[n].className != "megaDropdown" && n < mdd.length ) n++;

						//alert ( 0 - obj.offsetTop );
						var offset = 0 - obj.offsetTop - 1;
						mdd[n].style.top = offset + "px";						
					}
				}
			}
