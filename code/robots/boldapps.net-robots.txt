  $(function() {
    var $back_to_top_control = $('<div><i class="fa fa-angle-up"></i></div>')
        .attr({
          'id': "back-to-top-control",
          'class': "back-to-top-control",
          'title': "Scroll Back to Top"
        }).appendTo('body')
        .click(function(evt) {
          evt.preventDefault();
          $('html, body').animate({ scrollTop: 0}, 500);
          return false;
        });

    var detectScrollPosition = function() {
      if ($(window).scrollTop() > 100) {
        $back_to_top_control.fadeIn();
      } else {
        $back_to_top_control.fadeOut();
      }
    };

    detectScrollPosition();

    $(window).scroll(detectScrollPosition);

  });
</script>

        $target, toggleOtherServices;

    if ($otherToggle.length > 0) {

      if ($target) {

        var $textarea = $($target.find("textarea"));
        $textarea.data('dirty-value', $textarea.val());

        toggleOtherServices = function(evt) {
          if ($otherToggle.prop("checked") === true) {
            $target.slideDown(150);

            $target.slideUp(150);
            $textarea.data('dirty-value', $textarea.val());
            $textarea.val('');
          }
        };

        $otherToggle.on("click", toggleOtherServices);
        toggleOtherServices();
      }
    }
  })(window, document, jQuery);
