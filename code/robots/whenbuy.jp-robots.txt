	function formatJPY(price){

		var arrNum=srcPrice.split("");
		var begIndex=arrNum.length%3;
		var j=0;
		for(var i=0;i<arrNum.length;i++,j++){

				//searchUrl=searchRootUrl + "/ref";
				//searchUrl=searchRootUrl;  
				
				var srcPath=window.location.pathname;
				var srcSearch=window.location.search;

						|| srcPath.indexOf("/zh_TW/")!=-1 || srcPath.indexOf("/en_US/")!=-1 ){

				}
				return;
			}else{
				searchUrl=searchRootUrl + "/s/search-alias/" + searchIndex + "/topSellers" ;
			}
		}
