	$('#SearchForm').submit(function(){
		if($('#kw').val()!='') {
		window.location.href='/sponsor/?kw='+encodeURIComponent($('#kw').val());
		}
	return false;
	});

	$('#SearchForm button.register').click(function(){
		if($('#SearchForm select option:selected').val()!='') {
			window.location.href='/register/?ride_id=' + $('#SearchForm select option:selected').val();
		}
		else {
		alert("please select a ride from the list");
		}
	return false;
	});

	$('#SearchForm button.search').on('click',function(){

		if (!$(this).parent().hasClass('open')) {
			$(this).parent().addClass('open');
		}

		if($('#kw').val()!='') {
		window.location.href='/sponsor/?kw='+encodeURIComponent($('#kw').val());
		}
		return false;
	});

	$('#SearchForm button.close').on('click',function(){

		if ($(this).parent().hasClass('open')) {
			$(this).parent().removeClass('open');
		}
		return false;
	});
