		new WOW().init(); // init wow script
		
		// change nav bar style on scroll
		$(window).scroll(function(e){
			var distanceY = $(window).scrollTop(),
				shrinkOn = 50,
				brand = $(".myNavbar .navbar-brand"),
				links = $(".myNavbar .navbar-nav");
			if (distanceY > shrinkOn) {
				brand.addClass("smaller");
				links.addClass("smaller");
			} else {
				if (brand.hasClass("smaller")) {
					brand.removeClass("smaller");
					links.removeClass("smaller");
				}
			}
		});
		
		$('.followUs>.net').mouseover(function(){
			$(this).children('.info').show();
		}).mouseout(function(){
			$(this).children('.info').hide();
		});
		
		var lefty = setTimeout(function(){
			// Left side of the screen
			$('.lefty').mouseover(function(){
				$('#pubBg>.bgLeft').css('background-color','#c0392b');
			}).mouseout(function(){
				$('#pubBg>.bgLeft').removeAttr('style');
			});
			
			// Right side of the screen
			$('.righty').mouseover(function(){
				$('#pubBg>.bgRight').css('background-color','#7f8c8d');
			}).mouseout(function(){
				$('#pubBg>.bgRight').removeAttr('style');
			});
			
			clearTimeout(lefty);
		},600);
	</script>
