        function getQuerystring(key) {
            key = key.replace(/[\[]/, "\\\[").replace(/[\]]/, "\\\]");
            var regex = new RegExp("[\\?&]" + key + "=([^&#]*)");
            var qs = regex.exec(window.location.href);
            var rtnVal = '';
            if (qs != null)
                rtnVal = qs[1];

            return rtnVal;
        }

        function GotoSearchPage(searchtext) {
            var url;
            var defaultText = 'enter your search phrase';

            url = '/search.aspx';

            //alert(url);

            if (searchtext.length == 0 || searchtext.trim().length == 0 || searchtext == defaultText) {
                alert('Please enter a valid search term');
                return false;
            }
            else {

                url = url + "?search=" + escape(searchtext);
                url = url + "&sitesearch=1";

                UpdatePage(url);
            }
        }

                location.href = url;

            return false;
