var isOpinion = false;

/**
 * 意見を登録する
 *
 * @param string str 意見内容
 * @return void
 */
function goikening(str) {
	var suc  = false;
	var comp = false;

	if (isOpinion !== false) {
		alert('ただいまご意見を受付しております。しばらくお待ちください。');
		return ;
	}

	if (!str) {
		alert('メッセージを入力してから[意見を投稿する]ボタンをクリックしてください。');
		return ;
	}

	isOpinion = true;

	$.ajax({
		url     : "/api/opinion/save",
		async   : false,
		data    : {
			"opinion" : str, 
			"path"    : ""
		},
		type    : "POST",
		success : function(result) {
			if (result.status == 1) {
				suc = true;
			}
			else {
				alert(result.message);
			}
		},
		complete : function(tmp, message) {
			if (suc) {
				$('.js-opinion-form').hide();
				$('.js-opinion-finish').show();
			}
			isOpinion = false;
		}
	});

}

$(function(){
	/**
	 * 「意見を投稿する」クリック

		$("#goikenCommonArea").fadeIn("fast");
	})
	/**
	 * 閉じるクリック
	 * @return void
	 */
	$(".goikenClose").click(function(){
		$("#TB_overlay").fadeOut("fast").remove();
		$("#goikenCommonArea").hide();
	})
});
