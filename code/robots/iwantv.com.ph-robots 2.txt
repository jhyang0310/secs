	{
		// toggle loader and carousel display

		if (newScrWidth > 768) {
			if (newScrWidth ) {

			}
		}

			method: 'GET',
			success: function(response)
			{
				// if window  screen has been resized 
				// destroy slick slider and remove slides
				if (newScrWidth) {
					$('#carousel-banner').slick('unslick');
					$('#carousel-banner').slick('removeSlide', null, null, true);
					$carousel_banner.html(response);
					
					// call create slick function
					createSlick();
				} else {
					$carousel_banner.html(response);

					// call create slick function
					createSlick();
				}
			}
		})
	}

	// Check if window screen size resized then return -
	// the current screen size to get neccessary API content
	function resized(newScrWidth) {
		var scr_width = null;
		if (newScrWidth) {
			return scr_width = newScrWidth;
		} else {
			return scr_width = $(window).width();
		}
	}

		function getFn(sectionName)
		{
			var arrSectionName = ['latest-section','popular-section','livestream-section','world-section','sponsored-section','kapamilya-section'];
			var arrFn = ['get_recently_added','get_most_watched','get_livestreams','get_all_worlds','get_sponsored_worlds','get_all_shows'];
			return arrFn[arrSectionName.indexOf(sectionName)];
		}
		
		function loadSection(fn, $section){
			$section.show();
			$.ajax({
				url: 'tv/'+fn,
				method: 'GET',
				success: function(response)
				{
					$section.html(response);
					loadedSectionCount++;
					makeMobileReponsive();
					if(loadedSectionCount === max_section)
				    {
						$('#section-wrap > div > section').each(function(i){
							if(i%2)
							{
							   $(this).addClass('section-bg-color');
							}
						});
				    }
				}
			});
		}
