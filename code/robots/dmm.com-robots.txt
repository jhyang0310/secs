function setEventCM(adframe){
    var videoCmTag = $('#video-'+ adframe);
    var videoEventIndex = rectangleCM[adframe+"_videoIndex"];
    var videoData = [];
    var videoplayTag = $('#top-cm-div-token-'+ adframe);
    videoCmTag.get(0).addEventListener("ended", function() {
        //次(※通常枠は必ず同じもの)の動画を再生する処理
        endNextCM(adframe);
    });
    videoCmTag.get(0).addEventListener('canplaythrough', function(){
        //再生準備完了まで止め画像を表示
        videoplayTag.css('visibility', 'hidden');
    });
    if(rectangleCM["MOVIE_DATA"][adframe][0]){
        videoData = rectangleCM["MOVIE_DATA"][adframe][0];
    }else{
        videoData = rectangleCM["MOVIE_DATA"][adframe];
    }
    //videoCmTag.get(0).poster = videoData.posterBanner;
    videoCmTag.get(0).play();
    videoCmTag.get(0).muted = true;
    var videoBannerTag = $('#cm-banner-'+ adframe);
    videoBannerTag.hover(
        function(){
            if(adframe == "sv-1"){
                videoEventIndex = rectangleCM["sv-1_videoIndex"];
                $(this).find('img').attr('src',rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].onMouseBanner);
            }else if(adframe == "sv-2"){
                videoEventIndex = rectangleCM["sv-2_videoIndex"];
                $(this).find('img').attr('src',rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].onMouseBanner);
            }else if(adframe == "nv-1" || adframe == "nv-2"){
                $(this).find('img').attr('src',videoData.onMouseBanner);
            }
        },
        function(){
            if(adframe == "sv-1"){
                videoEventIndex = rectangleCM["sv-1_videoIndex"];
                $(this).find('img').attr('src',rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].cmBanner);
            }else if(adframe == "sv-2"){
                videoEventIndex = rectangleCM["sv-2_videoIndex"];
                $(this).find('img').attr('src',rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].cmBanner);
            }else if(adframe == "nv-1" || adframe == "nv-2"){
                $(this).find('img').attr('src',videoData.cmBanner);
            }
        }
    );
}

function endNextCM(adframe){
    var videoEventIndex;
    var videoCount= rectangleCM[adframe + "_LENGTH"];
    if(adframe === "sv-1"){
        rectangleCM["sv-1_videoIndex"]++;
        if(rectangleCM["sv-1_videoIndex"] > videoCount - 1) {rectangleCM["sv-1_videoIndex"] = 0};
        videoEventIndex = rectangleCM["sv-1_videoIndex"];
    }else if(adframe === "sv-2"){
        videoCount = rectangleCM["MOVIE_DATA"]["sv-2"].length;
        rectangleCM["sv-2_videoIndex"]++;
        if(rectangleCM["sv-2_videoIndex"] > videoCount - 1) {rectangleCM["sv-2_videoIndex"] = 0};
        videoEventIndex = rectangleCM["sv-2_videoIndex"];
    }else if(adframe === "nv-1" || adframe === "nv-2") {
        videoEventIndex = 0;
    }

    var video = $('#video-' + adframe);
    var videoLink = $('#movie-' + adframe);
    var cmBanner = $('#cm-banner-' + adframe);
    var cmSound = $('#cm-sound-token_' + adframe);

    function loopVideo() {
        var loadedDef = new $.Deferred();
        if(adframe === "sv-1" || adframe === "sv-2"){
            video.get(0).src = rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].videoBanner;
            cmBanner.find("img").attr('src', rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].cmBanner);

            videoLink.find("a").attr("target",rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].target);
            videoLink.find("a").attr("alt",rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].alt);

            // 止め絵の表示
            cmBanner.css('visibility', 'hidden');
            cmSound.css('visibility', 'hidden');

            video.get(0).load();
            loadedDef.resolve();
            return loadedDef.promise();
        }else if(adframe === "nv-1" || adframe === "nv-2"){
            video.get(0).src = rectangleCM["MOVIE_DATA"][adframe].videoBanner;
            cmBanner.find("img").attr('src', rectangleCM["MOVIE_DATA"][adframe].cmBanner);

            videoLink.find("a").attr("target",rectangleCM["MOVIE_DATA"][adframe].target);
            videoLink.find("a").attr("alt",rectangleCM["MOVIE_DATA"][adframe].alt);

            // 止め絵の表示
            cmBanner.css('visibility', 'hidden');
            cmSound.css('visibility', 'hidden');

            video.get(0).load();
            loadedDef.resolve();
            return loadedDef.promise();
        }
    };
    loopVideo()
        .then(
            function() {
                var timeDef = new $.Deferred();
                setTimeout(function () {
                    timeDef.resolve();
                }, 3000);
                return timeDef.promise();
            }
        ).then(
            function() {
                // 動画再生
                cmBanner.css('visibility', '');
                cmSound.css('visibility', '');
                if(adframe === "sv-1" || adframe === "sv-2") {
                    //video.get(0).poster = rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].posterBanner;
                    videoLink.find("a").attr('href', rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].clickUrl);
                    cmBanner.find("a").attr('href', rectangleCM["MOVIE_DATA"][adframe][videoEventIndex].clickUrl);
                }else{
                    //video.get(0).poster = rectangleCM["MOVIE_DATA"][adframe].posterBanner;
                    videoLink.find("a").attr('href', rectangleCM["MOVIE_DATA"][adframe].clickUrl);
                    cmBanner.find("a").attr('href', rectangleCM["MOVIE_DATA"][adframe].clickUrl);
                }
                video.get(0).play();
            }
    );
}

var soundOff ="http://p.dmm.com/p/top/bnr/sound_off.gif";
function createRectangleVideoDOM(ad, adframe)
{
	//動画ロード完了前に中心に表示
    var playImg = "http://p.dmm.com/p/top/bnr/bt_play.png";

    //指定した枠の一番上の動画広告
    adframe.indexOf("sv") != -1 ? ad = ad[adframe][0] : ad = ad[adframe];

    var sbRectElement = $("<div></div>", {"class": "sb rect-banner"});
    var sbsElement = $("<div></div>", {"class": "sbs"});
    var wrapElement = $("<div></div>", {"class": "wrap-cm"});
    var areaElement = $("<div></div>", {"class": "area-cm"});
    var videoplayElement = $("<div></div>", {"class": "top-cm","id":'top-cm-div-token-'+ adframe, position:'absolute', top:'0', left:'0', width: 244, height: 165});
    var contentsElement = $("<div></div>", {"class": "contents-cm"});
    var movieElement = $("<p></p>", {"class": "movie", "id": "movie-" + adframe});
    var videoLink = $("<a></a>", {href: ad.clickUrl});
    if(adframe.indexOf("sv") != -1){
        var videoElement = $("<video></video>", {src: ad.videoBanner, id: "video-" + adframe, preload: true, muted: true, controls: false, "class": "is-play"});
    }else{
        var videoElement = $("<video></video>", {src: ad.videoBanner, id: "video-" + adframe, preload: true, muted: true, controls: false, "class": "is-play" ,loop:true});
    }

    videoElement.on("contextmenu",function(){return false;});
    videoLink.append(videoElement);
    movieElement.append(videoLink);

    var soundElement = $("<span></span>", {"class": "cm-sound", "id": "cm-sound-token_" + adframe});
    soundElement.append($("<img></img>", {src: soundOff, width: '62', height: '12'}));

    var cmBannerElement = $("<p></p>", {"class": "cm-banner", id: "cm-banner-" + adframe});
    var cmBannerhref = $("<a></a>", {href: ad.clickUrl}).append($('<img />', {src: ad.cmBanner, width: 244, height: 30}));
    cmBannerElement.append(cmBannerhref);

    contentsElement.append(movieElement).append(soundElement);
    areaElement.append(contentsElement).append(cmBannerElement);

    var playElement =$("<p></p>").append($('<img />', {src: playImg, width: 50, height: 104, alt:""}));
    videoplayElement.append(playElement);
    wrapElement.append(areaElement);
    wrapElement.append(videoplayElement);

    sbsElement.append(wrapElement);
    sbRectElement.append(sbsElement);

    //各種ハンドラの設定
    //動画の再生
    videoplayElement.click(function(){
        videoElement.play();
        videoplayElement.css('visibility', 'hidden');
        cmSound.find("img").css('visibility', '');
    });

    //音声ミュート
    soundElement.click(function () {
        videoElement.get(0).muted = !videoElement.get(0).muted;
        //デフォルトに反映
        videoElement.get(0).defaultMuted = videoElement.get(0).muted;
        //ミュート状態で画像を差し替え
        var fn = 'sound_on.gif';
        if (videoElement.get(0).muted) {
            fn = 'sound_off.gif';
        }
        $(this).find('img').attr('src', 'http://p.dmm.com/p/top/bnr/' + fn);
    });
    soundElement.hover(
        function () {
            $(this).css("cursor", "pointer");
        },
        function () {
            $(this).css("cursor", "default");
        }
    );
    videoElement.on('ended', function () {
        $(this).addClass('is-ended');
        $(this).removeClass('is-play');
    });
    videoElement.on('play', function () {
        $(this).removeClass('is-ended');
        $(this).addClass('is-play');
    });

    return sbRectElement;
}
