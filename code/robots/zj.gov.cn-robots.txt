function ShowTag(TagName,TagClass,num,count,more) {

		var tag = document.getElementById(TagName+"_tab_"+i);
		var cont = document.getElementById(TagName+"_cont_"+i);
		var mores;

				if(more) {
			mores.style.display = "none";
			} 
				}
		}
