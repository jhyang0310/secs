<script>
  function count_down(id)
  {
      var hd =0;  var md =0;
      var sec= document.getElementById("sec"+id);
      var min= document.getElementById("min"+id);
      var hor= document.getElementById("hor"+id);

      var counter_down = setInterval( function()
      {
          var dtnow = new Date;
          if( (dtnow.getMinutes() == "00" && dtnow.getSeconds() == "00") || (dtnow.getMinutes() == "0" && dtnow.getSeconds() == "0") )
          {
            get_tb("home_tbl","panel-2","tod");
          }
          else if( hor.innerHTML == 0 && min.innerHTML == 0 && sec.innerHTML == 0 )
          {
              get_tb("home_tbl","panel-2","tod");
              clearInterval(counter_down);
          }

          else if( hor.innerHTML > 0 && hd ==0 && md ==0 )
          {
              if( min.innerHTML == 0 )
              {
                  if( sec.innerHTML == 0 )
                  {
                      hor.innerHTML=hor.innerHTML*1-1;
                      min.innerHTML=59;
                      sec.innerHTML=59;
                  }
                  else
                  {
                      sec.innerHTML=sec.innerHTML*1-1;
                  }
              }
              else if( min.innerHTML > 0 )
              {
                  if( sec.innerHTML == 0 )
                  {
                      min.innerHTML=min.innerHTML*1-1;
                      sec.innerHTML=59;
                      md=1;
                  }
                  else
                  {
                      sec.innerHTML=sec.innerHTML*1-1;
                  }
              }
          }

          else if( hor.innerHTML == 0 && hd ==0 && md ==0 )
          {
              if( min.innerHTML > 0 )
              {
                  if( sec.innerHTML == 0 )
                  {
                      min.innerHTML=min.innerHTML*1-1;
                      sec.innerHTML=59;
                      md=1;
                  }
                  else
                  {
                      sec.innerHTML=sec.innerHTML*1-1;
                  }
              }
              if( min.innerHTML == 0 )
              {
                  if( sec.innerHTML == 0 )
                  {
                      min.innerHTML=99;
                  }
                  else
                  {
                      sec.innerHTML=sec.innerHTML*1-1;
                  }
              }
          }

          else
          {
              if( sec.innerHTML > 0 )
              {
                  sec.innerHTML=sec.innerHTML*1-1;
              }
              else if( sec.innerHTML == 0 )
              {
                  hd=0; md=0;
              }
          }



      } , 1000 )

  }

<script>
function send_form(thisObj,form_name,id=0,alert_all="",old) // in cp and site send froms like new post , new pck ..etc
{
    setTimeout( function()
    {
      var formData = new FormData($("form#"+form_name)[0]);
      formData.append("form",form_name);
      formData.append("form_id",id);

      if(form_name == "page_new" || form_name == "post_new" || form_name == "news_new")
      {
        var des = CKEDITOR.instances.cke_new.getData(); formData.append("des",des);
      }
      if(form_name.search("page_edit_") > 0)
      {
        var des = CKEDITOR.instances["cke_edit"+id].getData(); formData.append("des",des);
      }

      $.ajax({
        url: 'ajax.php', type: 'POST',data: formData, async: false,cache: false, contentType: false, processData: false,success: function (returndata)
        {
          if(alert_all != ""){ alert(form_name+"--"+returndata); }
          var info = jQuery.parseJSON(returndata);
          if (info.type == "TRUE")
          {
            if(info.do=="reset")
            {
              try { $('#'+form_name)[0].reset(); }
              catch(err) { try { CKEDITOR.instances.cke_new.destroy(true); $('#'+form_name)[0].reset(); } catch(err){} }
              try {
                    $('.select2-hidden-accessible').each(function()
                    {
                      $('select').select2();
                      $(this).select2("val", $(this).val());
                      $('select').select2();
                    });
                  }
             catch(err){}
            }

            // $('.modal').modal('hide');
            swal("", info.msg, "success");
          }
          else { swal("", info.msg, "error"); }


          try { eval(info.do); }
          catch(err) { try{ do_now(info.do); } catch(err){} }

        }
      });
      $(thisObj).prop("disabled",false);
      $(thisObj).html(old);

    }, 1379 )
}


</script>

<script>

var progress_bar_timer;
var i = 0;

  function go_to(value) // in cp go to part of admin like posts , news , upload ...etc
  {
    clearTimeout(progress_bar_timer);
    i=0; progress_bar_timer = setInterval( function() { progress_bar_plus1(value) }, 100);
  }

  function progress_bar_plus1(value)
  {
      $('#progress_bar').attr('aria-valuenow',i ).css('width',i +'%' );
      if(i==50) { document.getElementById("main_content").innerHTML =""; }
      else if(i==70) { progress_bar_plus2(value); }
      else if(i==100)
      {
        i= 0;
        $('#progress_bar').attr('aria-valuenow',i ).css('width',i +'%' );
        $('select').select2();
        //if(value == "mth"){ get_tb("mth","tab2"); }
        if(value == "channels"){ get_tb("channels","tab2"); }
        clearInterval(progress_bar_timer) ;
        return;
      }
      i=i+5*1;
  }
  function progress_bar_plus2(value)
  {
    clearTimeout(progress_bar_timer);
    var xhttp = new XMLHttpRequest();
    xhttp.onreadystatechange = function()
    {
      if (xhttp.readyState == 4 && xhttp.status == 200)
      {
        document.getElementById("main_content").innerHTML = xhttp.responseText;
        progress_bar_timer = setInterval( function() { progress_bar_plus1(value) }, 100);
      }
    }
    xhttp.open("POST", "ajax.php", true); xhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
    xhttp.send("get=go_to&id="+value);
  }
</script>
