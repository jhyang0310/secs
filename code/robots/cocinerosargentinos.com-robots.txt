        function validarEmail(email) {
    		var re = /^(([^<>()\[\]\\.,;:\s@"]+(\.[^<>()\[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
    		return re.test(email);
    	}

        function enviarRegistro() {
        	if(registroPendiente) return;
        	
        	registroPendiente = true;
        	
        	var nombre = $("#registro-nombre"),
        		email = $("#registro-email"),
        		provincia = $("#registro-provincia"),
        		pais = $("#registro-pais"),
        		password = $("#registro-password"),
        		repetirPassword = $("#registro-repetir-password");
        	
        	if(nombre.val().trim().length < 2) {
        		alert("El nombre debe tener mas de 2 caracteres");
        		nombre.focus();
        		registroPendiente = false;
        		return;
        	}
        	
        	if(provincia.val().trim().length < 2) {
        		alert("La provincia debe tener mas de 2 caracteres");
        		provincia.focus();
        		registroPendiente = false;
        		return;
        	}
        	
        	if(pais.val().trim().length < 2) {
        		alert("El país debe tener mas de 2 caracteres");
        		pais.focus();
        		registroPendiente = false;
        		return;
        	}
        	
        	if(!validarEmail(email.val())) {
        		alert("El email es inválido");
        		email.focus();
        		registroPendiente = false;
        		return;
        	}
        	
        	if(password.val().length < 6) {
        		alert("La contraseña debe tener 6 o mas caracteres");
        		password.focus();
        		registroPendiente = false;
        		return;
        	}
        	
        	if(password.val() != repetirPassword.val()) {
        		alert("Las contraseñas no coinciden");
        		repetirPassword.focus();
        		registroPendiente = false;
        		return;
        	}
        	
        	if(!$("#acepta-terminos").is(":checked")) {
        		alert("Debe leer y aceptar los términos y condiciones");
        		registroPendiente = false;
        		return;
        	}
        	
        	//TODO: Crear la variable URL de document
        	$.ajax({
        		url: "ajax/registrarse.ajax.php?accion=registro",
        		data: "nombre=" + escape(nombre.val()) + "&email=" + escape(email.val()) + "&provincia=" + escape(provincia.val()) + "&pais=" + escape(pais.val()) + "&password=" + escape(password.val()) + "&repetir-password=" + escape(repetirPassword.val()),
        		type: "post",
        		dataType: "json",
        		complete: function() {
        			registroPendiente = false;
        		},
        		success: function(data) {
        			if(data) {
        				if(data.resultado == false) {
        					alert(data.error);
        				} else {
        					alert("Muchas gracias por registrarse");
        					window.location = window.location.href;
        				}
        			}
        		}
        	});
        }
