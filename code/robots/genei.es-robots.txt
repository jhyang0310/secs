    function entrarse()
    {
        $('#entrarse').attr("disabled", "true");
        $("#entrarse").css('background-image', 'url("https://www.genei.es/img/ajax-loader.svg")');
        $("#entrarse").css('background-repeat', 'no-repeat');
        datos = {
            'usr': $("#usr").val(),
            'pwd': $("#pwd").val(),
            'captcha_usr': $("#captcha_usr").val()
        };
        $.ajax({
            url: base_url + 'usuarios/acceso',
            type: 'post',
            cache: false,
            data: datos,
            success: function (data) {
                window.location.href = base_url + 'usuarios/miusuario';
                return true;

            },
            error: function (e, obj) {
                alert("Error en llamada -> " + e);
                return false;
            }

        });
    }
    function registrarse()
    {
        if ($("#condiciones_registro").is(':checked'))
            var condiciones_registro = 'si';
        else
            var condiciones_registro = false;
        if ($('#tipo_registro_empresa').is(':checked'))
            tipo_registro = 'empresa';
        else
            tipo_registro = 'particular';
        datos = {
            'tipo_registro': tipo_registro,
            'nombre_registro': $("#nombre_registro").val(),
            'apellidos_registro': $("#apellidos_registro").val(),
            'nif_cif_registro': $("#nif_cif_registro").val(),
            'telefono_registro': $("#telefono_registro").val(),
            'email_registro': $("#email_registro").val(),
            'direccion_registro': $("#direccion_registro").val(),
            'poblacion_registro': $("#poblacion_registro").val(),
            'prevision_envios_registro': $("#prevision_envios_registro").val(),
            'codigo_postal_registro': $("#codigo_postal_registro").val(),
            'id_pais_registro': $("#id_pais_registro").val(),
            'provincia_registro': $("#provincia_registro").val(),
            'usr_registro': $("#usr_registro").val(),
            'pwd_registro': $("#pwd_registro").val(),
            'padrino': $("#padrino").val(),
            'condiciones_registro': condiciones_registro

        };
        //llamamos a traves de ajax al controlador con los datos para que guarde la direccion.
        $.ajax({
            url: base_url + 'usuarios/registro_modal',
            type: 'post',
            cache: false,
            data: datos,
            success: function (data) {
                //$('#div_precio').html(data);
                //alert(data);
                if (data == 'ok')
                {
                    $('#modal_registrar').modal('hide');
                    $('#modal_registro_ok').modal('show');
                } else
                {
                    $('#modal_errores_registro .modal-body').html(data);
                    $('#modal_errores_registro').modal('show');
                    //recargamos la pagina para que actualice la capa del popup con los nuevos datos.
                    //location.reload();
                    return false;
                }

            },
            error: function (e, obj) {
                alert("Error en llamada -> " + e);
                return false;
            }

        });
        return false;
    }

        function comprobar_nif_nie(value)
        {
            var validChars = 'TRWAGMYFPDXBNJZSQVHLCKET';
            var nifRexp = /^[0-9]{8}[TRWAGMYFPDXBNJZSQVHLCKET]{1}$/i;
            var nieRexp = /^[XYZ]{1}[0-9]{7}[TRWAGMYFPDXBNJZSQVHLCKET]{1}$/i;
            var str = value.toString().toUpperCase();

            if (!nifRexp.test(str) && !nieRexp.test(str))
                return false;

            var nie = str
                    .replace(/^[X]/, '0')
                    .replace(/^[Y]/, '1')
                    .replace(/^[Z]/, '2');

            var letter = str.substr(-1);
            var charIndex = parseInt(nie.substr(0, 8)) % 23;

            if (validChars.charAt(charIndex) === letter)
                return true;

            return false;
        }
