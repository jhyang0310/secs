jQuery(document).ready(function($) {
	var totalPages = 292;
	var search = '';
	var url = 'https://www.dexerto.com/';
	var scrollHandling = {
		allow: true,
		reallow: function() {
			scrollHandling.allow = true;
		},
		delay: 400 //(milliseconds) adjust to the highest acceptable value
	};
	var button = $('.load-more');
		
	var doingAjax = false;
	var lastLoadedPage = 0; 
	var page = 2; // First request == page 2
		
	$('body').on('click','.load-more',function(e){
		if  ( (totalPages > 1 && !doingAjax) && (lastLoadedPage !== totalPages) ){

	$(window).scroll(function(){
		if( ! doingAjax && scrollHandling.allow ) {

			scrollHandling.allow = false;
			setTimeout(scrollHandling.reallow, scrollHandling.delay);
			var offset = $(button).offset().top - $(window).scrollTop();
			if( 2000 > offset ) {
				console.log('Load Article triggered');
				loadArticle(page, button, search);
			}
		}
	});
	
	function loadArticle(pageNumber, btn, search=null)
	{
		doingAjax = true;

		lastLoadedPage = page;
		btn.addClass('loading');
		var paged = 'paged';
		$.ajax({
			url: url + '?' + paged + '=' + pageNumber + '&s=' + search,
			type:'GET',
			dataType:'html',
			success: function(html){
				var items = $(html).find('#three-content-col').html();
				$('.dx_body').trigger( "archive:paged", [items] );

				if (page == totalPages) {
					btn.remove();
					$(document.body).trigger("sticky_kit:recalc");
				} else {
					btn.removeClass('loading');
				}
				page++;
				doingAjax = false;
			}

		});
		return false;
	}
});
