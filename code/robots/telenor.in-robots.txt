    jQuery(document).ready(function() {

    function showSuggest() {
        var url = 'https://suggestqueries.google.com/complete/search?';
        url += 'callback=JSON_CALLBACK&client=firefox&hl-en&q=';
        url += jQuery("#auto_search").val();
        jQuery.ajax({
            url: url,
            dataType: 'JSONP',
            headers: {
                'Access-Control-Allow-Origin': '*',
                'Access-Control-Allow-Methods': '*',
                'Content-Type': 'text/plain'
            },
            success: function(response) {
                jQuery('#display').show();
                var jj = String(response);
                var final_data = jj.split(',');
                var fullhtml = "<ul style='list-style: none;' id='auto_search_data_ul_Id'>";
                for (var j in final_data) {
                    var fullhtml = fullhtml + "<li onclick='fill(\"" + final_data[j] + "\")'>" + final_data[j] + "</li>";
                }
                fullhtml = fullhtml + "</ul>"
                document.getElementById('display').innerHTML = fullhtml;
            }
        });
    }
    function fill(Value) {
        jQuery('#auto_search').val(Value);
        jQuery('#display').hide();
    }
