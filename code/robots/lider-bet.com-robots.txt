    function numberIt(num) {
        var jack = num.toString().split('');
        var l = jack.length;
        var j = '';
        for (var k = 0; k < l; k++) {
            //$('.pn_num_' + i).html(jackpot[i]);
            if (k == l - 2) {
                j += '.';
                j += jack[k];
            } else {
                j += jack[k];
            }
        }
        return jack;
    }

    function checkZero(num){
	    if(num == 0){
	        num = 0.01;
        }else{
            num = num.toString().split('.');
            if(num[1] == '00' || num[1].indexOf('0') == 1){
                num = num[0]+'.'+num[1];
                num = parseFloat(num);
                num = num + 0.01;
            }else{
                num = num[0]+'.'+num[1];
            }

        }
        return num;
    }

	function BannerbetTimeOut() {

    function tslider(action) {
        var ban = $('.top-match-wrap .top-match');
        var activeBan = $('.top-match-wrap .top-match.active');
        var banIndex;
        clearInterval(bannerbetAutoPlay);

        ban.removeClass('active nextleft nextright prevleft prevright');

        switch(action){
            case 'next':
                banIndex = activeBan.next().index() == -1? 0:activeBan.next().index();
                var nextleft = banIndex == 0? ban.length:banIndex;

                ban.eq(nextleft-1).addClass('nextleft');
                ban.eq(banIndex).addClass('active nextright');
                activeBannerID = ban.eq(banIndex).attr('data-id');
                break;
            case 'prev':
                banIndex = activeBan.prev().index() == -1? ban.length-1 : activeBan.prev().index();
                var prevleft = banIndex == ban.length-1? -1:banIndex;

                ban.eq(prevleft+1).addClass('prevleft');
                ban.eq(banIndex).addClass('active prevright');
                activeBannerID = ban.eq(banIndex).attr('data-id');
                break;
        }
        BannerbetTimeOut();
    }


	function setBannerbetAutoPlay() {
        tslider('next')
    }
