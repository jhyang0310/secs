                            <p><strong>解决方案：</strong>根据学生的成绩，依靠裸分的话只可以考虑广州大学、海南大学、河北医科大学这些普通类院校。但通过与学生的沟通了解，从小学习过德语，以后想出国。并且动手能力较强。经过合理定位，为他推荐了哈尔滨工程大学车辆工程类专业（哈尔滨工程大学是211院校，往年的录取分数都在620分左右）。然后通过指导写论文、撰写自荐信的方式通过了该校的的自招初审环节。在高考前成功获得该校30分加分的资格。</p>
                            <br>
                            <p><strong>录取结果：</strong>通过自招加分，成功被哈尔滨工程大学的自动化专业录取。</p>
                        </dd>

                            <br>

    var lunbo=(function(params){
        var MobileArea = function() {
            this.$key=0;
        }
        MobileArea.prototype = {
            init:function(params){
                this.params=params;
                this.element1=params.element1;
                this.element=params.element;
                this.speed=params.speed;
                this.active=params.active;
                this.bindEvent();
            },
            bindEvent: function(){
                var _self = this;
                var timer;
                _self.element.click(function(event) {
                clearInterval(timer);    
                timer=setInterval(autoplay, 3000);
                $(this).addClass(_self.active).siblings().removeClass(_self.active);
                _self.element1.eq($(this).index()).stop(true,true).fadeIn(_self.speed).siblings().stop(true,true).fadeOut(_self.speed);
                _self.$key=$(this).index(); 
                }); 
                timer=setInterval(autoplay, 3000); 
                _self.element1.hover(function() {
                   clearInterval(timer);
                }, function() {
                    clearInterval(timer);
                    timer=setInterval(autoplay, 3000); 
                });
                function autoplay(){
                    _self.$key++;         
                    if(_self.$key>2) 
                    {
                        _self.$key=0;   
                    }
                    _self.element.eq(_self.$key).addClass(_self.active).siblings().removeClass(_self.active);
                    _self.element1.eq(_self.$key).stop(true,true).fadeIn(_self.speed).siblings().stop(true,true).fadeOut(_self.speed);
                }
            }
        }
        return MobileArea;
    })()
