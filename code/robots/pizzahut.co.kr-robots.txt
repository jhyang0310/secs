	$(".header_top_search_close").click(function(e){
		$("input[name=cmmn_top_search]").val("");
	});

	
	$("#m_cmmn_top_search_btn").next().click(function(e){
		$("input[name=m_cmmn_top_search]").val("");
	});

	
	$(".header_top_search_btn").click(function(e){
		var str = $("input[name=cmmn_top_search]").val();
		if(str.replace(/\s/g, "") == "") {
			alert("검색어를 입력해주세요.");
			return false;
		}
		fn_searchFormSubmit(str);
		return false;
	});

	
	$("#m_cmmn_top_search_btn").click(function(e){
		var str = $("input[name=m_cmmn_top_search]").val();
		if(str.replace(/\s/g, "") == "") {
			alert("검색어를 입력해주세요.");
			$(this).prev().focus();
			return false;
		}
		fn_searchFormSubmit(str);
		return false;
	});

	
	$("input[name=m_cmmn_top_search]").keyup(function(e){
		if (e.keyCode == 13) {
			$("#m_cmmn_top_search_btn").click();
			return false;
		}
	}).keydown(function(e){
		$(this).keyup();
	});

	
	$("input[name=cmmn_top_search]").keyup(function(e){
		if (e.keyCode == 13) {
			$(".header_top_search_btn").click();
			return false;
		}

		$(".header_top_search_auto a").text("");
		var str = $(this).val().replace(/\s/g, "");
		if(str == "") return false;
		var array = _pizzaNmList;
		var viewArray = [];
		for (var i = 0; i < array.length; i++) {
			var pNm = array[i].pizzaName.replace(/\s/g, "");
			if(pNm.indexOf(str) == 0) {
				viewArray.push(array[i].pizzaName);
				if(viewArray.length >= 10){
					break;
				}
			}
		}

		for (var i = 0; i < 10; i++) {
			if(typeof viewArray[i] != 'undefined'){
				$(".header_top_search_auto a").eq(i).text(viewArray[i]);
			}else{
				$(".header_top_search_auto a").eq(i).html("&nbsp;");
			}
		}
	}).keydown(function(e){
		$(this).keyup();
	});

	
	$(".header_top_search_auto a").click(function(e){
		var str = $(this).text();
		if(str.replace(/\s/g, "") == "") { return false; }
		fn_searchFormSubmit($(this).text());
		return false;
	});

	
	function fn_searchFormSubmit(str){
		var url = '/common/search';
		var form = $('<form action="' + url + '" method="post">').appendTo(document.body);
		form.append($('<input name="pzNm"/>').val(str));
		form.submit();
	}

	var callBackFn_sideBoxSelected;
	function fn_open_sideBox(cid, bid, pid, sid, cnt, callBack){
		callBackFn_sideBoxSelected = callBack;
		// 사이드박스 제품이 존재한다면..
		callAjaxRequestBody(
			"/menu/pizza/action/sideboxListCnt",
			ajaxRequest({cid:cid, bid:bid, pid:pid, sid:sid, maxCnt:cnt}),
			function (oo) {
				if (!oo.success){ if (oo.message) { alert(oo.message); } else { alert('사이드박스의 정보 요청에 실패 하였습니다.'); } return false; }
				var result = oo.data;
				if($.isNumeric(result) && result >= 1){
					// 사이드 박스 선택 레이어 팝업 오픈
					document.getElementById("Pop_Common").innerHTML = "";
					var param = "?cid=" + cid + "&bid=" + bid + "&pid=" + pid + "&sid=" + sid + "&maxCnt=" + cnt;
					lp_open('/menu/pizza/popup/sideBox' + param, popupOpenInitSideBox);
				}else{
					callBack([]);
				}
			},
			function (fal) {
				alert("사이드박스의 정보 요청에 실패 하였습니다.");
			}
		);
	}

	function popupOpenInitSideBox(){
		var tabLength = $(".slider_ul").length;

		if (tabLength < maxTabNum) {
			maxTabNum = tabLength;
		}
		// Tab Product 및 Open 메뉴활성화
		var numTotal = 0;
		var tabIndex = 0;
		numTotal++;
		setTabProduct(tabIndex);
		lp_renew();
	}
