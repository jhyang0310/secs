    $(function(){

        $('#form_trial2').bind('submit', function(event){
            event.preventDefault();

            var params = {};
            var inputs = $(this).find('input, select');
            for (var i = 0; i < inputs.length; i++) {
                var element = inputs.eq(i);

                if (element.attr('type') == "checkbox") {
                    params[element.attr('name')] = $(element).get(0).checked ? "on" : "off";
                } else {
                    if ($.trim(element.val()) == "") {
                        element.closest('.shake-container').effect('shake', { distance : 4, times : 2, duration : 500});
                        element.focus();
                        return;
                    }

                    if (element.attr('data-validation') == "email" && !Nitro.utils.Validator.validEmail($.trim(element.val()))) {
                        element.closest('.shake-container').effect('shake', { distance : 4, times : 2, duration : 500});
                        element.focus();
                        return;
                    }

                    params[element.attr('name')] = $.trim(element.val());
                }
            }

            $(this).find('.loader').show();
            $(this).find('input, select').attr('readonly', true).addClass('disabled');
            $(this).find('button').addClass('disabled');

            console.log(params);

            $.ajax({
                type : "POST",
                url : "/nitroconversionservices/trialmodalform",
                data: params,
                complete: function() {
                                        window.location = 'https://www.gonitro.com/try/pdftoppt/download/thanks';

            });
        });
    });
