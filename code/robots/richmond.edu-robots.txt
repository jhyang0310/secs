                function resizeImg(img, callback) {
                    callback = callback || false;

                    if(typeof callback !== "function"){
                        callback = false;
                    }

                    if(img.length != "undefined" && img.length == 1){
                        img = img[0];
                    }

                    var breakpoints = new Array();
                    var attributes = img.attributes;

                    for (var i = 0; i < attributes.length; ++i) {
                        if (attributes[i].name.indexOf("data-img") >= 0) {
                            breakpoints.push(parseInt(attributes[i].name.replace(/\D/g, "")));
                        }
                    }

                    breakpoints.sort(function (a, b) {
                        return a - b
                    }); //orders breakpoints from lowest to highest

                    var size = breakpoints[0]; //size defaults to smallest value
                    var sizeTarget = img.getElementWidth();

                    //resizeIgnore class name forces this function to look up the parental tree one level to gain its size target
                    function getParentWidth(parent){
                        if(parent.offsetWidth != 0 && !parent.classList.contains("resizeIgnore")){
                            sizeTarget = parent.getElementWidth();
                        } else {
                            parent = parent.parentNode;
                            getParentWidth(parent);
                        }
                    }

                    if(sizeTarget == 0 || img.parentNode.classList.contains("resizeIgnore")){
                        var parent = img.parentNode;
                        getParentWidth(parent);
                    }

                    for (var i = 0; i < breakpoints.length; i++) {
                        if (sizeTarget > breakpoints[i]) {
                            if (i == breakpoints.length - 1) { //last breakpoint
                                size = breakpoints[i];
                            } else {
                                size = breakpoints[i + 1];
                            }
                        }
                    }

                    var hasResponsiveImage = img.getElementsByClassName("img-" + size).length;

                    if (hasResponsiveImage == 0) {
                        var src = img.getAttribute("data-img-" + size);
                        var alt = img.getAttribute("data-alt");
                        var imageCredit = "";

                        if (img.getElementsByClassName("imageCredit").length > 0) {
                            imageCredit = img.getElementsByClassName("imageCredit")[0].innerText || img.getElementsByClassName("imageCredit")[0].textContent;
                        }

                        img.innerHTML = '<img src="' + src + '" alt="' + alt + '" class="img-' + size + '" />';
                        if (imageCredit != "") {
                            img.innerHTML = img.innerHTML + '<span class="imageCredit">' + imageCredit + '</span>';
                        }
                    }

                    if(callback !== false){
                        callback();
                    }
                }

                function resizeImagery(){
                    var images = document.getElementsByClassName("responsiveImg");
                    for(var i=0; i<images.length; i++){
                        resizeImg(images[i]);
                    }
                }

                if (window.addEventListener) {
                    window.addEventListener("resize", resizeImagery);
                }
                else {
                    window.attachEvent("resize", resizeImagery);
                }
