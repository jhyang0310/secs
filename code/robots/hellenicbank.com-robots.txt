		/* Net Banking Form validation and actions*/
		function validateNetBankingForm(){
			var login_type=document.getElementById('netBankingLoginType').value;
			
			if(login_type=='personal'){
				//action should go to personal
				if (document.netBankingloginForm.ps_subscriber_id.value == "") {
				    alert(login_subscriber_id_required);
			    	return(false); 
				}
				
				else if (IsNumeric(document.netBankingloginForm.ps_subscriber_id.value) == false){
			      alert(login_subscriber_id_number);
				  return(false);
			    }
				else if (document.netBankingloginForm.ps_password.value == "") {
				    alert(login_password_required);
				    return(false);
				}
				else{
					document.getElementById('subscriber').value=document.getElementById('ps_subscriber_id').value;
					document.getElementById('username').value=document.getElementById('ps_subscriber_id').value;
					document.getElementById('password').value=document.getElementById('ps_password').value;
					document.netBankingloginForm.submit();
					/* clears the form after submission*/
					clearNetBankingElement('loginTabPersonal');
				}
			}
			else{
				//action should go to business
				if (document.netBankingloginForm.bs_subscriber_id.value == "") {
				    alert(login_subscriber_id_required);
				    return(false);
				}
				else if (IsNumeric(document.netBankingloginForm.bs_subscriber_id.value) == false){
			      alert(login_subscriber_id_number);
				  return(false);
			    }
				else if (document.netBankingloginForm.bs_business_id.value == "") {
				    alert(login_business_id_required);
			    	return(false); 
				}
				else if (IsNumeric(document.netBankingloginForm.bs_business_id.value) == false){
			      alert(login_business_id_number);
				  return(false);
			    }
				else if (document.netBankingloginForm.bs_password.value == "") {
				    alert(login_password_required);
				    return(false);
				}
				else{
					document.getElementById('subscriber').value=document.getElementById('bs_subscriber_id').value;
					document.getElementById('password').value=document.getElementById('bs_password').value;
					document.getElementById('business').value=document.getElementById('bs_business_id').value;
					document.getElementById('username').value=document.getElementById('subscriber').value+document.getElementById('business').value;
					document.netBankingloginForm.submit();
					/* clears the form after submission*/
					clearNetBankingElement('loginTabBusiness');
				}
			}

		}
