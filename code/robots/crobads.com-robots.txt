<script>

$(document).ready(function() {

    //*** ANIMATION NUMBER

    jQuery(function() {
        var isStartNumbers = false;
        var numbers = $('#facts-numbers');
        $(window).scroll(function () {
            if (isStartNumbers) return;
            var numbersOffset = numbers.offset().top + numbers.height() * 0.1;
            var scrollOffset = $(window).scrollTop() + $(window).height();


            if (scrollOffset < numbersOffset) {
                return;
            }
            isStartNumbers = true;
            numbers.find('.value span').each(function() {
                var $this = $(this);
                var isDecimal = $this.hasClass('is-decimal');
                $this.animateNumber({
                    number: $this.attr('num'),
                    numberStep: function(now) {
                        now = Math.floor(now);
                        if (isDecimal) {
                            now /= 10;
                            now = now.toString();
                            if (now.indexOf('.') == -1) {
                                now += '.0';
                            }
                        }
                        $this.text(now);
                    }
                }, 3000);
            });

        });

    });

    jQuery(function() {
        var isStartNumbers = false;
        var string = $('#animate_string');
        $(window).scroll(function () {
            if (isStartNumbers) return;
            // var numbersOffset = string1.offset().top + string1.height() * 0.1;
            var numbersOffset = string.offset().top + string.height() * 0.1;
            // var numbersOffset1 = string3.offset().top + string3.height() * 0.1;
            var scrollOffset = $(window).scrollTop() + $(window).height();
            if (scrollOffset < numbersOffset) {
                return;
            }
            isStartNumbers = true;
            $('.animate_string').addClass('animated bounceInRight');
            $('.h1_string').addClass('animated bounce');

        });


    });


});
