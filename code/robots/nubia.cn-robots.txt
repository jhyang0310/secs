<script type="text/javascript" tag="user scripts">
    $(function(){
        //carousel

        $('.input-group').on('focus', '.form-control', function () {
            $(this).closest('.input-group, .form-group').addClass('focus');
        }).on('blur', '.form-control', function () {
            $(this).closest('.input-group, .form-group').removeClass('focus');
        });

        //Email checked
        var posting = false;
        $('#mc-embedded-subscribe').on('click', function() {
            var $this = $(this);
            var emailVal = $('#mce-EMAIL').val();
            var emailreg = /^(\w)+(\.\w+)*@(\w)+((\.\w+)+)$/;
            if(!emailreg.test(emailVal)) {
                $('#email-msg-wrong').fadeIn();
                setTimeout(function() {
                    $('#email-msg-wrong').fadeOut();
                }, 1000);
                return false
            }
            if(posting) { return false; }
            $.ajax({
                type: 'POST',
                url: '?a=subscribe',
                dataType: 'json',
                data: {
                    'email': emailVal
                }
            }).done(function() {
                $('#email-msg-right').fadeIn();
                setTimeout(function() {
                    posting = false;
                    $('#email-msg-right').fadeOut();
                    $('#mce-EMAIL').val('');
                }, 1000);
            });
        });

        $("#swiper a.btn-video,#small_ad a.btn-video").on('click', function(e){
            e.preventDefault();
            var href = $(this).attr('href'),
                title = $(this).find('h3').text(),
                $modal = $('#videoBanner');
            console.log(title);

            $modal.find('video').attr('src', href);
            $('#ModalLabel').text(title);
            $modal.modal();
        })

    })
</script>
