	                for (var i = 0; i < cookieArray.length; i++) {
	                    var cookie = cookieArray[i].split('='),
	                        name = cookie[0],
	                        value = cookie[1];
	
	                    if (name === cookieName) {
	                        return value;
	                    }
	                }
	                return null;
	            }
	
	            function stringifyCookieObject(cookieObject) {
	                return encodeURIComponent(JSON.stringify(cookieObject));
	            }
	
	            function addCookieObject(name, object, options) {
	                document.cookie = [
	                    encodeURIComponent(name), '=', stringifyCookieObject(object),
	                    '; path=' + options.path,
	                    '; domain=' + options.domain,
	                ].join(''); 
	            }
	
	            function insertLoader() {
	                var loaderElementClass = document.getElementById('first-overlay').className,
	                    regex = new RegExp('(?:^|\\s)'+ 'skin-0081' +'(?:\\s|$)');
	
	                if (!regex.test(loaderElementClass)) {
	                    document.getElementById('first-overlay').className += ' skin-0081';
	                }
	            }
	
	            function removeLoader() {
	                var loaderElement = document.getElementById('first-overlay'),
	                    regex = new RegExp('(?:^|\\s)'+ 'skin-0081' +'(?:\\s|$)');
	
	                loaderElement.className = loaderElement.className.replace(regex, '');
	            }
	
	            function setTudoAzulAndRedirect(data) {
	                if (data) {
	                    addCookieObject(
	                        'TudoAzul',
	                        data,
	                        { path: '/', domain: azulSettings.Domain }
	                    );
	                    window.location.reload();
	                }
	                else {
	                    removeLoader();
	                }
	            }
	
	            function requestProfileAndRedirect(token) {
	                var xhr = new XMLHttpRequest();
	
	                xhr.open('POST', azulSettings.Services.POST_LOGIN);
	                xhr.setRequestHeader('Content-Type', 'application/x-www-form-urlencoded');
	
	                xhr.onreadystatechange = function() {
	                    if (xhr.readyState === XMLHttpRequest.DONE) {
	
	                        if (xhr.status === 200) {
	                            var jsonData = JSON.parse(xhr.responseText);
	                            setTudoAzulAndRedirect(jsonData.data.ReturnObject);
	                        }
	                        else {
	                            removeLoader();
	                        }
	                    }
	                };
	
	                xhr.send(encodeURI('SkySalesToken=' + token));
	            }
	
	            if (!document.querySelector('#home-logged')) {
	
	                var skySalesToken = getCookieValue('SkySalesToken'),
	                    tudoAzul = getCookieValue('TudoAzul');
	
	                if (skySalesToken !== null) {
	                    if (tudoAzul == null) {
	                        insertLoader();
	                        requestProfileAndRedirect(skySalesToken);
	                    }
	                }
	            }
	        })();
