  function fetch_article(url, disable_related){
    $.ajax({
      method: 'GET',
      url: url,
      beforeSend: function() {
      	// scroll to top
      	// $(window).scrollTop(0);
      	// clear previous content
      	$('#article-holder .content').html('');
      	//freezes body
      	body_paralyzed();
      	// pop
		   	$('#article-holder').addClass('pop-open loading');
		   	if (disable_related){$('#article-holder').addClass('disable-related');}

		  }
    }).done(function( data ) {
    	// fetch and put in the content
    	window.history.pushState('', '', url);
    	$('#article-holder').removeClass('loading');

    	ga('set', 'page', '/s/'+$(data).find('#article').data('article-id'));

			ga('set', 'dimension1', $(data).find('#article').data('article-id'));
			ga('set', 'dimension2', $(data).find('#article').data('article-section-id'));


			ga('set', 'dimension3', $(data).find('#article').data('article-title'));
			ga('set', 'dimension4', $(data).find('#article').data('article-section-title'));

			ga('set', 'dimension5', $(data).find('#article').data('article-channel-id'));
			ga('set', 'dimension6', $(data).find('#article').data('article-channel-title'));

			ga('send', 'pageview');


      $('#article-holder .content').html($(data).find('#fetch').html());
      initiate_article();

    });
  }



  function fetch_related(entry_id){
	  $.ajax({
	    method: 'GET',
	    url: document.location.protocol + '//www.themalaysianinsight.com/s/related/' + entry_id,
	    beforeSend: function() {}}).done(function( data ) {
	  	if ((!$("div.news-item[data-news='"+entry_id+"']").next().hasClass('related-read')) && ($('.news-item[data-related-news="'+entry_id+'"]').length == 0)){
	  		console.log('newly opend');
	  	}


	  	if ((!$("div.news-item[data-news='"+entry_id+"']").next().hasClass('related-read')) || ($('.news-item[data-related-news="'+entry_id+'"]').length == 0)){$('.related-read').remove();}	  	

	  	if (!$("div.news-item[data-news='"+entry_id+"']").next().hasClass('related-read')){
	  		$("div.news-item[data-news='"+entry_id+"']").after(data);	

	  		$('#close-related').click(function() {$('.related-read').remove();});
	  	}

	  });
	}


	function	initiate_comments(){
		$('.comments .reply').click(function() {
		  $('#comment_form input[name=parent_id]').val($(this).attr('rel'));
		  $('#comment_form').insertAfter( // Insert the comment form after...
		  $(this).parent() // The containing li tag
		  );
		  $('#comment_form').show();
		});
		$('.comments .quote').click(function() {
		  $('#comment_form textarea[name=comment]').val('[quote]'+ 
		  $(this).parent().parent().find('.comment-content').text()+
		  '[/quote]'
		  );
		});

		$('#comment_form').on('submit', function(e){

		  e.preventDefault();

		  console.log('form submits');

		  $.ajax({
		    method: "POST",
		    url: $('#comment_form').attr('action'),
		    data: $('#comment_form').serialize()
		  }).done(function( data ) {
		    $('.comments-box').remove();
		    fetch_comments($('div#article').data('article-id'));

		    
		  });

		});
	}

	function fetch_comments(entry_id){
	  $.ajax({
	    method: 'GET',
	    url: document.location.protocol + '//www.themalaysianinsight.com/comments/' + entry_id,
	    beforeSend: function() {}}).done(function( data ) {
	  	$('#load-comment').hide();
	  	$('div#article div.single').after(data);
	  	initiate_comments();
	  });
	}
